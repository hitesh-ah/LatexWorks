\question
The state-space representation of a system is given
by:\\
$
\boldsymbol{\dot{x}}=
\begin{bmatrix}
-1 & 0\\
0 & -2
\end{bmatrix} 
x(t)+
\begin{bmatrix}
1 \\
1 
\end{bmatrix}
u(t), y(t) = 
\begin{bmatrix}
1 & 1\\
\end{bmatrix} 
x(t)
\ \ \ 
$
\\The transfer function of this system is
\begin{enumerate}[label=(\Alph*)]
\item ${(s^2+3s+2)}^{-1}$
\item ${(s+2)}^{-1}$
\item $s{(s^2+3s+2)}^{-1}$
\item ${(s+1)}^{-1}$
\end{enumerate}
\begin{solution}
(D)
\\ $T(s)=
\begin{bmatrix}
1 \\
1 
\end{bmatrix}
{(s\boldsymbol{I}-\boldsymbol{A})^{-1}}
\begin{bmatrix}
1 \\
0 
\end{bmatrix}\\
(s\boldsymbol{I}-\boldsymbol{A})^{-1}=
\begin{bmatrix}
\frac{1}{s+1} & 0\\
0 & \frac{1}{s+2}
\end{bmatrix} 
\\T(s)=
\begin{bmatrix}
1 \\
1 
\end{bmatrix}
\begin{bmatrix}
\frac{1}{s+1} & 0\\
0 & \frac{1}{s+2}
\end{bmatrix}
\begin{bmatrix}
1 \\
0 
\end{bmatrix}
=\frac{1}{s+1}
$
\end{solution}

\question
The phase margin of a system with the open loop transfer function:\\
$\displaystyle G(s)H(s)=\frac{(1-s)}{(1+s)(3+s)}$ is
\begin{enumerate}[label=(\Alph*)]
\item $68.3^{\circ}$
\item $90^{\circ}$
\item $0^{\circ}$
\item $\infty$
\end{enumerate}
\begin{solution}
(D)\\
$|{GH(j\omega)|\neq 1}$, for any value of $\omega$. Thus phase margin is $\infty$
\end{solution}
